let balance = Number(localStorage.getItem("balance")) || 1000; // SAFE ELEMENT GETTER const el = (id) => document.getElementById(id); // REGISTER function register() { let u = el("username").value; let p = el("password").value; if (!u || !p) return alert("Fill all fields"); localStorage.setItem("user", u); localStorage.setItem("pass", btoa(p)); // simple encoding alert("Account created!"); } // LOGIN function loginUser() { let u = el("username").value; let p = el("password").value; if ( u === localStorage.getItem("user") && btoa(p) === localStorage.getItem("pass") ) { startApp(); } else { alert("Wrong details"); } } // START APP function startApp() { el("login").style.display = "none"; el("app").classList.remove("hidden"); el("user").innerText = localStorage.getItem("user"); updateBalance(); loadPrice(); loadChart(); } // NAV function show(page) { ["dashboard", "trade", "chart", "wallet"].forEach((id) => { el(id).classList.add("hidden"); }); el(page).classList.remove("hidden"); } // BALANCE function updateBalance() { el("balance").innerText = balance; el("walletBalance").innerText = balance; localStorage.setItem("balance", balance); } // SAFE ACTIONS function deposit() { balance += 100; updateBalance(); } function withdraw() { if (balance >= 50) balance -= 50; updateBalance(); } function buy() { if (balance >= 100) { balance -= 100; el("tradeMsg").innerText = "Bought BTC"; } else { el("tradeMsg").innerText = "Not enough balance"; } updateBalance(); } function sell() { balance += 100; el("tradeMsg").innerText = "Sold BTC"; updateBalance(); } // PRICE function loadPrice() { fetch("https://api.coingecko.com/api/v3/simple/price?ids=bitcoin&vs_currencies=usd") .then((r) => r.json()) .then((d) => { el("btcPrice").innerText = d.bitcoin.usd; }); } // CHART function loadChart() { fetch("https://api.coingecko.com/api/v3/coins/bitcoin/market_chart?vs_currency=usd&days=7") .then((r) => r.json()) .then((d) => { let prices = d.prices.map((p) => p[1]); new Chart(el("chartCanvas"), { type: "line", data: { labels: prices.map((_, i) => i), datasets: [{ data: prices, label: "BTC Price" }], }, }); }); } // LOGOUT function logout() { localStorage.clear(); location.reload(); } // AUTO LOGIN if (localStorage.getItem("user")) startApp();
